--BikeToDrive_Fiets_Verkoop;

CREATE TABLE Klant (
	klantnr INT,
	naam VARCHAR(45),
	adres VARCHAR(100),
	woonplaats VARCHAR(45),
	geslacht VARCHAR(10),
	geboortedatum DATE,

	PRIMARY KEY(klantnr)
);

CREATE TABLE Filiaal (
	filiaalnr INT,
	naam VARCHAR(45),
	adres VARCHAR(100),
	provincie VARCHAR(45),

	PRIMARY KEY(filiaalnr)
);

CREATE TABLE Monteur (
	monteurNr INT,
	naam VARCHAR(45),
	woonplaats VARCHAR(45),
	uurloon FLOAT,
	filiaal INT NOT NULL,

	PRIMARY KEY(monteurNr),
	FOREIGN KEY(filiaal) REFERENCES Filiaal(filiaalnr)
);

CREATE TABLE Fabrikant (
	fabrikantnr INT,
	naam VARCHAR(45),
	adres VARCHAR(100),
	plaats VARCHAR(45),

	PRIMARY KEY(fabrikantnr)
);

CREATE TABLE Fiets (
	fietsnr INT,
	soort VARCHAR(45),
	merk VARCHAR(45),
	type VARCHAR(45),
	standaardprijs FLOAT,
	inkoopprijs FLOAT,
	fabrikant INT NOT NULL,

	PRIMARY KEY(fietsnr),
	FOREIGN KEY(fabrikant) REFERENCES Fabrikant(fabrikantnr)
);

CREATE TABLE Fiets_Verkoop (
	verkoopnr INT,
	datum DATE,
	aantal INT,
	verkoopprijs FLOAT,
	klant INT NOT NULL,
	fiets INT NOT NULL,
	monteur INT NOT NULL,

	PRIMARY KEY(verkoopnr),
	FOREIGN KEY(klant) REFERENCES Klant(klantnr),
	FOREIGN KEY(fiets) REFERENCES Fiets(fietsnr),
	FOREIGN KEY(monteur) REFERENCES Monteur(monteurNr)
);
